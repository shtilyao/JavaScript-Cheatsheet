01

Переменные

Основы
const
Постоянное значение — не переприсваивается
const name = 'Аня';
const PI = 3.14;
// name = 'Маша'; — ошибка!
Используй const по умолчанию. Объекты и массивы через const — их содержимое можно менять.
let
Переменная — можно переприсваивать
let count = 0;
count = 5;  // ок
count++;   // count = 6
Если значение нужно менять — используй let. Видна только внутри своего блока {}.
var
Устаревший способ — почти не используется
var x = 1;  // избегай
⚠ var не имеет блочной области видимости и «поднимается». Используй const/let.

Операторы

Основы
Арифметические
10 + 3   // 13
10 - 3   // 7
10 * 3   // 30
10 / 3   // 3.333...
10 % 3   // 1 (остаток)
2 ** 8   // 256 (степень)
x++      // x = x + 1
x--      // x = x - 1
Сравнение
5 === 5    // true  ← используй это
5 !== 6    // true
'5' == 5   // true  ← приводит типы!
5 > 3     // true
5 < 3     // false
5 >= 5    // true
5 <= 4    // false
⚠ Всегда используй === вместо ==
Логические
true && false   // false (И)
true || false   // true  (ИЛИ)
!true           // false (НЕ)

// Короткое замыкание:
user && user.name  // name только если user есть
name || 'Аноним'  // 'Аноним' если name falsy
name ?? 'Аноним'  // только если null/undefined
Присваивание
let x = 10;
x += 5;    // x = 15
x -= 3;    // x = 12
x *= 2;    // x = 24
x /= 4;    // x = 6
x %= 4;    // x = 2
x **= 3;   // x = 8
x ??= 5;   // если x = null/undefined → 5
Тернарный оператор
Краткий if/else
const msg = age >= 18
  ? 'взрослый'
  : 'ребёнок';

// Вложенный (лучше избегать):
const grade = score >= 90 ? 'A'
  : score >= 70 ? 'B' : 'C';
04

Условия

Основы
if / else if / else
if (score >= 90) {
  console.log('Отлично');
} else if (score >= 70) {
  console.log('Хорошо');
} else {
  console.log('Попробуй ещё');
}
switch
Когда вариантов много
switch (day) {
  case 'пн':
  case 'вт':
    console.log('Рабочий');
    break;
  case 'сб':
  case 'вс':
    console.log('Выходной');
    break;
  default:
    console.log('Не знаю');
}
Не забывай break — иначе выполнятся все следующие case.
05

Циклы

Основы
for
Классический цикл со счётчиком
for (let i = 0; i < 5; i++) {
  console.log(i); // 0 1 2 3 4
}

// С шагом 2:
for (let i = 0; i < 10; i += 2) { ... }

// Обратный:
for (let i = 4; i >= 0; i--) { ... }
for...of
Перебор элементов массива/строки
const fruits = ['яблоко', 'груша'];

for (const fruit of fruits) {
  console.log(fruit);
}

// С индексом:
for (const [i, val] of fruits.entries()) {
  console.log(i, val);
}
for...in
Перебор ключей объекта
const user = { name: 'Аня', age: 25 };

for (const key in user) {
  console.log(key, user[key]);
  // 'name' 'Аня'
  // 'age' 25
}
⚠ Не использовать for...in для массивов.
while / do...while
let i = 0;
while (i < 3) {
  console.log(i++);
}

// do...while — выполняется хотя бы раз:
do {
  console.log(i);
} while (i < 0);
break / continue
for (let i = 0; i < 5; i++) {
  if (i === 2) continue; // пропустить 2
  if (i === 4) break;    // выйти на 4
  console.log(i); // 0 1 3
}
06

Функции

Основы
Объявление функции
Function declaration — поднимается (hoisting)
function greet(name) {
  return `Привет, ${name}!`;
}

greet('Аня'); // 'Привет, Аня!'
Стрелочная функция
Краткий синтаксис, нет своего this
// Полная запись:
const add = (a, b) => {
  return a + b;
};

// Короткая (один return):
const add = (a, b) => a + b;

// Один параметр — скобки не нужны:
const double = x => x * 2;
Параметры по умолчанию
function greet(name = 'Мир', greeting = 'Привет') {
  return `${greeting}, ${name}!`;
}

greet();          // 'Привет, Мир!'
greet('Аня');     // 'Привет, Аня!'
Callback (функция-аргумент)
Передаём функцию в другую функцию
function doTwice(fn) {
  fn();
  fn();
}

doTwice(() => console.log('привет'));
// 'привет'
// 'привет'
Замыкание (closure)
Функция «запоминает» своё окружение
function counter() {
  let count = 0;
  return () => ++count;
}

const c = counter();
c(); // 1
c(); // 2
c(); // 3
Внутренняя функция имеет доступ к переменным внешней даже после её завершения.
IIFE
Функция, которая вызывает себя сразу
(function() {
  console.log('запустилась сразу');
})();

// Или со стрелкой:
(() => {
  console.log('и эта тоже');
})();

Массивы — методы

Основы
Добавление и удаление
const arr = [1, 2, 3];

arr.push(4);    // [1,2,3,4] — в конец
arr.pop();      // [1,2,3] — убрать с конца
arr.unshift(0); // [0,1,2,3] — в начало
arr.shift();    // [1,2,3] — убрать с начала
Поиск
const arr = [1, 2, 3, 4];

arr.includes(3)          // true
arr.indexOf(3)           // 2
arr.find(x => x > 2)    // 3 (сам элемент)
arr.findIndex(x => x > 2) // 2 (индекс)
arr.some(x => x > 3)    // true (хотя бы один)
arr.every(x > 0)         // true (все)
.map()
Создать новый массив, преобразовав каждый элемент
const nums = [1, 2, 3];

nums.map(x => x * 2);
// [2, 4, 6]

nums.map((x, i) => `${i}: ${x}`);
// ['0: 1', '1: 2', '2: 3']
Не изменяет исходный массив. Всегда возвращает массив той же длины.
.filter()
Оставить только подходящие элементы
const nums = [1, 2, 3, 4, 5];

nums.filter(x => x > 2);
// [3, 4, 5]

nums.filter(x => x % 2 === 0);
// [2, 4] — только чётные
.reduce()
Свернуть массив в одно значение
const nums = [1, 2, 3, 4];

// Сумма:
nums.reduce((acc, x) => acc + x, 0);
// 10

// Максимум:
nums.reduce((max, x) => x > max ? x : max, -Infinity);
Второй аргумент — начальное значение аккумулятора.
Остальные важные методы
const arr = [3, 1, 4, 1];

arr.slice(1, 3)       // [1, 4] — не меняет
arr.splice(1, 2)      // меняет! удаляет 2 с позиции 1
arr.sort((a,b) => a-b) // [1,1,3,4] — меняет!
arr.reverse()         // меняет!
arr.join(', ')       // '3, 1, 4, 1'
arr.flat(Infinity)   // разгладить вложенные
Array.from('abc')    // ['a','b','c']
Array.isArray(arr)  // true

Объекты

Основы
Создание и доступ
const user = {
  name: 'Аня',
  age: 25,
  greet() {
    return `Привет, я ${this.name}`;
  }
};

user.name         // 'Аня'
user['name']     // 'Аня'
user.greet()     // 'Привет, я Аня'
user.city        // undefined
Object методы
const obj = { a: 1, b: 2, c: 3 };

Object.keys(obj)    // ['a', 'b', 'c']
Object.values(obj) // [1, 2, 3]
Object.entries(obj) // [['a',1],['b',2],['c',3]]

// Перебор объекта:
Object.entries(obj).forEach(([k, v]) => {
  console.log(k, v);
});
Опциональная цепочка и ??
const user = { address: null };

// Без ?. — ошибка!
user.address.city;  // TypeError

// С ?. — безопасно:
user?.address?.city; // undefined

// Значение по умолчанию:
user?.name ?? 'Аноним'; // 'Аноним'
Копирование и слияние
const a = { x: 1 };
const b = { y: 2 };

// Spread (поверхностная копия):
const copy = { ...a };
const merged = { ...a, ...b }; // {x:1, y:2}

// Object.assign:
Object.assign({}, a, b); // {x:1, y:2}

// Глубокая копия (простых объектов):
JSON.parse(JSON.stringify(obj));

Деструктуризация

Средний
Деструктуризация массива
const [a, b, c] = [1, 2, 3];
// a=1, b=2, c=3

// Пропустить элемент:
const [, second] = [1, 2, 3];
// second = 2

// Со значением по умолчанию:
const [x = 10] = [];
// x = 10

// Swap переменных:
[a, b] = [b, a];
Деструктуризация объекта
const user = { name: 'Аня', age: 25 };

const { name, age } = user;
// name='Аня', age=25

// Переименование:
const { name: n } = user;
// n='Аня'

// Значение по умолчанию:
const { city = 'Неизвестно' } = user;
// city='Неизвестно'
В параметрах функции
// Без деструктуризации:
function greet(user) {
  return user.name;
}

// С деструктуризацией:
function greet({ name, age = 0 }) {
  return `${name}, ${age} лет`;
}

greet({ name: 'Аня', age: 25 });
Вложенная деструктуризация
const data = {
  user: {
    name: 'Аня',
    address: { city: 'Киев' }
  }
};

const { user: { name, address: { city } } } = data;
// name='Аня', city='Киев'

Spread / Rest оператор ...

Средний
Spread — развернуть
// Массивы:
const a = [1, 2];
const b = [3, 4];
const all = [...a, ...b]; // [1,2,3,4]
const copy = [...a];      // [1,2]

// Объекты:
const user = { name: 'Аня' };
const full = { ...user, age: 25 };

// В вызове функции:
Math.max(...[1, 5, 3]); // 5
Rest — собрать остаток
// В функции:
function sum(...nums) {
  return nums.reduce((a,b) => a+b, 0);
}
sum(1, 2, 3, 4); // 10

// В деструктуризации:
const [first, ...rest] = [1, 2, 3, 4];
// first=1, rest=[2,3,4]

const { name, ...other } = { name:'Аня', age:25, city:'Киев' };
// name='Аня', other={age:25, city:'Киев'}

Классы

Средний
Базовый класс
class Animal {
  // Конструктор — вызывается при new
  constructor(name, sound) {
    this.name = name;
    this.sound = sound;
  }

  // Метод:
  speak() {
    return `${this.name} говорит ${this.sound}`;
  }
}

const cat = new Animal('Кот', 'мяу');
cat.speak(); // 'Кот говорит мяу'
Наследование
class Dog extends Animal {
  constructor(name) {
    super(name, 'гав'); // вызов родителя
    this.tricks = [];
  }

  learn(trick) {
    this.tricks.push(trick);
  }
}

const dog = new Dog('Рекс');
dog.speak();    // 'Рекс говорит гав'
dog.learn('сидеть');
Getter, Setter, Static
class Circle {
  constructor(r) { this.r = r; }

  // Getter — как свойство:
  get area() {
    return Math.PI * this.r ** 2;
  }

  // Статический метод:
  static compare(c1, c2) {
    return c1.r - c2.r;
  }
}

const c = new Circle(5);
c.area; // ~78.5 (без скобок!)
Circle.compare(c1, c2);
Приватные поля (#)
class BankAccount {
  #balance = 0; // приватное поле

  deposit(amount) {
    if (amount > 0) this.#balance += amount;
  }

  get balance() { return this.#balance; }
}

const acc = new BankAccount();
acc.deposit(100);
acc.#balance; // SyntaxError!

Промисы и async/await

Продвинутый
Promise
Обещание будущего результата
const p = new Promise((resolve, reject) => {
  if (ok) resolve('данные');
  else    reject(new Error('ошибка'));
});

p
  .then(data => console.log(data))
  .catch(err => console.error(err))
  .finally(() => console.log('конец'));
async / await
Синтаксический сахар над промисами
async function getData() {
  try {
    const res = await fetch('https://api.example.com');
    const data = await res.json();
    return data;
  } catch (err) {
    console.error('Ошибка:', err);
  }
}

getData().then(d => console.log(d));
⚠ Всегда оборачивай await в try/catch
Promise.all / allSettled
// Все или никто:
const [u, p] = await Promise.all([
  fetchUser(),
  fetchPosts()
]);

// Дождаться всех, даже если ошибки:
const results = await Promise.allSettled([...]);
results.forEach((r) => {
  if (r.status === 'fulfilled') ...
});
fetch API
// GET:
const res = await fetch(url);
const data = await res.json();

// POST:
await fetch(url, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ name: 'Аня' })
});

// Проверить статус:
if (!res.ok) throw new Error(res.status);

Продвинутые концепции

Продвинутый
Map и Set
// Map — словарь, ключи любого типа:
const map = new Map();
map.set('key', 42);
map.get('key');   // 42
map.has('key');   // true
map.delete('key');
map.size;          // 0

// Set — уникальные значения:
const set = new Set([1, 2, 2, 3]);
// Set(3) {1, 2, 3}
const unique = [...new Set(arr)]; // убрать дубли
Генераторы
function* range(start, end) {
  for (let i = start; i <= end; i++) {
    yield i;
  }
}

for (const n of range(1, 5)) {
  console.log(n); // 1 2 3 4 5
}

const gen = range(1, 3);
gen.next(); // {value: 1, done: false}
Symbol и итераторы
// Symbol — уникальный ключ:
const id = Symbol('id');
const obj = { [id]: 42 };
obj[id]; // 42

// Кастомный итератор:
const range = {
  [Symbol.iterator]() {
    let i = 0;
    return { next: () =>
      i < 3
        ? { value: i++, done: false }
        : { done: true }
    };
  }
};
Proxy и Reflect
const handler = {
  get(target, key) {
    return key in target
      ? target[key]
      : `Нет ключа "${key}"`;
  },
  set(target, key, val) {
    if (typeof val !== 'number') throw new Error('Только числа!');
    target[key] = val; return true;
  }
};
const p = new Proxy({}, handler);
WeakMap / WeakRef
// WeakMap — не мешает сборке мусора:
const wm = new WeakMap();
let obj = {};
wm.set(obj, 'данные');
obj = null; // 'данные' удалятся

// WeakRef — слабая ссылка:
const ref = new WeakRef(obj);
ref.deref(); // объект или undefined
Error — обработка ошибок
try {
  throw new Error('что-то пошло не так');
} catch (err) {
  console.log(err.message); // сообщение
  console.log(err.name);    // 'Error'
  console.log(err.stack);   // стектрейс
} finally {
  console.log('выполнится всегда');
}

// Свой тип ошибки:
class ValidationError extends Error {
  constructor(msg) { super(msg); this.name = 'ValidationError'; }
}
